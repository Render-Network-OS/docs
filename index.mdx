<script>
  // Hide Mintlify branding after page load
  document.addEventListener('DOMContentLoaded', function() {
    // Hide any elements containing Mintlify branding
    const elements = document.querySelectorAll('*');
    elements.forEach(element => {
      if (element.textContent && element.textContent.includes('Powered by Mintlify')) {
        element.style.display = 'none';
        element.style.visibility = 'hidden';
        element.style.opacity = '0';
        element.style.pointerEvents = 'none';
        element.style.width = '0';
        element.style.height = '0';
        element.style.overflow = 'hidden';
      }
      if (element.innerHTML && element.innerHTML.includes('mintlify')) {
        element.style.display = 'none';
      }
    });

    // Hide theme toggle buttons
    const themeButtons = document.querySelectorAll('button[aria-label*="theme"], button[aria-label*="Theme"]');
    themeButtons.forEach(button => {
      button.style.display = 'none';
    });

    // Continuous check for new elements
    const observer = new MutationObserver(function(mutations) {
      mutations.forEach(function(mutation) {
        mutation.addedNodes.forEach(function(node) {
          if (node.nodeType === 1) { // Element node
            const element = node as HTMLElement;
            if (element.textContent && element.textContent.includes('Powered by Mintlify')) {
              element.style.display = 'none';
            }
            if (element.innerHTML && element.innerHTML.includes('mintlify')) {
              element.style.display = 'none';
            }
          }
        });
      });
    });

    observer.observe(document.body, {
      childList: true,
      subtree: true
    });
  });
</script>
